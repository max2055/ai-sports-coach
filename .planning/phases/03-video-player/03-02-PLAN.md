## 标注绘制规范（参考 tennis_annotate.py）
- body: 蓝色圆圈 (#1E90FF)
- hand_l: 绿色圆圈 (#32C850) - 持拍手
- hand_r: 深绿色圆圈 (#148A32) - 非持拍手
- foot_l: 橙色圆圈 (#FF8C00) - 前脚
- foot_r: 深橙色圆圈 (#DC6400) - 后脚
- issue_ring: 红色外圈 (#FF3232) - 问题高亮
- good_ring: 绿色外圈 (#32DC50) - 正确标注

## Canvas 叠加实现

- 计算当前帧号: frameNumber = floor(time * fps)
- 查找该帧的标注数据
- 在 Canvas 上绘制圆圈和标签

    创建 `web/src/api/frames.ts`:
    ```typescript
    import axios from 'axios'
    import type { IssueType } from '../types/upload'

    const API_BASE = import.meta.env.VITE_API_BASE_URL || 'http://localhost:8000'

    export interface BodyPartAnnotation {
      x_pct: number
      y_pct: number
      radius_pct: number
      issue_type?: IssueType
      issue_note?: string
    }

    export interface PlayerAnnotation {
      body: BodyPartAnnotation
      hand_l: BodyPartAnnotation
      hand_r: BodyPartAnnotation
      foot_l: BodyPartAnnotation
      foot_r: BodyPartAnnotation
    }

    export interface FrameAnnotation {
      frame_number: number
      players: PlayerAnnotation[]
    }

    export interface IssueFrame {
      frame_number: number
      issue_types: IssueType[]
      thumbnail_url: string
      time_seconds: number
    }

    export async function getFrameAnnotations(videoId: string): Promise<FrameAnnotation[]> {
      const response = await axios.get(`${API_BASE}/api/frames/${videoId}`)
      return response.data
    }

    export async function getIssueFrames(videoId: string, issueType?: string): Promise<IssueFrame[]> {
      const url = issueType
        ? `${API_BASE}/api/issues/${videoId}?issue_type=${issueType}`
        : `${API_BASE}/api/issues/${videoId}`
      const response = await axios.get(url)
      return response.data
    }

    export function getAnnotatedImageUrl(videoId: string, frameNumber: number): string {
      return `${API_BASE}/api/annotated/${videoId}/${frameNumber}`
    }
    ```
