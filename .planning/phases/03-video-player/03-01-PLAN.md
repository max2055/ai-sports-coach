       ```python
       from pydantic import BaseModel
       from typing import Literal, Optional

       IssueType = Literal[
           "LATE BACKSWING", "WRONG GRIP", "GRIP CHANGE ERR", "WRONG CONTACT",
           "ELBOW DROP", "NO HIP ROT", "LATE SHOULDER", "NO FOLLOW-THRU",
           "POOR FOOTWORK", "NO SPLIT STEP", "LATE WEIGHT", "WRONG STANCE",
           "WRONG POSITION", "BAD TOSS", "NO LEG DRIVE", "OFF BALANCE",
           "TELEGRAPH", "WRONG SPIN", "GOOD FORM", "GOOD FOOTWORK", "GOOD TACTICS"
       ]

       class BodyPartAnnotation(BaseModel):
           x_pct: float
           y_pct: float
           radius_pct: float
           issue_type: Optional[IssueType] = None
           issue_note: Optional[str] = None

       class PlayerAnnotation(BaseModel):
           body: BodyPartAnnotation
           hand_l: BodyPartAnnotation  # 持拍手
           hand_r: BodyPartAnnotation  # 非持拍手
           foot_l: BodyPartAnnotation  # 前脚
           foot_r: BodyPartAnnotation  # 后脚

       class FrameAnnotation(BaseModel):
           frame_number: int
           players: list[PlayerAnnotation]

       class IssueFrame(BaseModel):
           frame_number: int
           issue_types: list[IssueType]
           thumbnail_url: str
           time_seconds: float
       ```

    1. 创建 `api/routers/frames.py`:
       ```python
       from fastapi import APIRouter, HTTPException
       from fastapi.responses import FileResponse
       from models.frame import FrameAnnotation, IssueFrame
       from services.frame_service import get_frame_annotations, get_issue_frames, get_annotated_image

       router = APIRouter()

       @router.get("/frames/{video_id}", response_model=list[FrameAnnotation])
       async def list_frame_annotations(video_id: str):
           return get_frame_annotations(video_id)

       @router.get("/issues/{video_id}", response_model=list[IssueFrame])
       async def list_issue_frames(video_id: str, issue_type: str = None):
           return get_issue_frames(video_id, issue_type)

       @router.get("/annotated/{video_id}/{frame_number}")
       async def get_annotated_frame(video_id: str, frame_number: int):
           path = get_annotated_image(video_id, frame_number)
           return FileResponse(path)
       ```
