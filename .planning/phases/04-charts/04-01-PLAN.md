    ```python
    from pydantic import BaseModel
    from typing import Literal

    class ServeHeightPoint(BaseModel):
        frame_number: int
        time_seconds: float
        height_pct: float  # body.y_pct (lower = higher height)

    class ServeHeightData(BaseModel):
        points: list[ServeHeightPoint]

    class HitPoint(BaseModel):
        frame_number: int
        x_pct: float
        y_pct: float
        issue_type: str | None

    class HitPointHeatmap(BaseModel):
        points: list[HitPoint]

    class RadarDimension(BaseModel):
        name: str
        value: int  # 0-10

    class ConsistencyRadar(BaseModel):
        dimensions: list[RadarDimension]
        overall_score: int

    class IssueStat(BaseModel):
        issue_type: str
        count: int
        severity: Literal["high", "medium", "low"]

    class IssueStatistics(BaseModel):
        stats: list[IssueStat]
        total_issues: int
        total_frames: int
    ```

  <action>
    1. 创建 `api/routers/charts.py`:
       ```python
       from fastapi import APIRouter
       from models.chart import ServeHeightData, HitPointHeatmap, ConsistencyRadar, IssueStatistics
       from services.chart_service import (
           get_serve_height_data, get_hit_point_heatmap,
           get_consistency_radar, get_issue_statistics
       )

       router = APIRouter()

       @router.get("/charts/{video_id}/serve-height", response_model=ServeHeightData)
       async def serve_height_chart(video_id: str):
           return get_serve_height_data(video_id)

       @router.get("/charts/{video_id}/hit-points", response_model=HitPointHeatmap)
       async def hit_points_chart(video_id: str):
           return get_hit_point_heatmap(video_id)

       @router.get("/charts/{video_id}/radar", response_model=ConsistencyRadar)
       async def radar_chart(video_id: str):
           return get_consistency_radar(video_id)

       @router.get("/charts/{video_id}/issues", response_model=IssueStatistics)
       async def issue_stats_chart(video_id: str):
           return get_issue_statistics(video_id)
       ```

    2. 更新 `api/main.py`:
