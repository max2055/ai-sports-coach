    ```typescript
    import axios from 'axios'

    const API_BASE = import.meta.env.VITE_API_BASE_URL || 'http://localhost:8000'

    export type ProVideoType = 'forehand' | 'backhand' | 'serve' | 'volley'

    export interface ProVideo {
      id: string
      type: ProVideoType
      name: string
      description: string
      file: string
      duration?: number
    }

    export interface ProVideoList {
      videos: ProVideo[]
    }

    export async function getProVideos(type?: ProVideoType): Promise<ProVideo[]> {
      const url = type
        ? `${API_BASE}/api/pros?type=${type}`
        : `${API_BASE}/api/pros`
      const response = await axios.get<ProVideoList>(url)
      return response.data.videos
    }

    export async function getProVideo(videoId: string): Promise<ProVideo> {
      const response = await axios.get<ProVideo>(`${API_BASE}/api/pros/${videoId}`)
      return response.data
    }

    export function getProVideoFileUrl(videoId: string): string {
      return `${API_BASE}/api/pros/${videoId}/file`
    }
    ```

    实现要点:
    ```vue
    <script setup lang="ts">
    import { ref, watch, onMounted } from 'vue'
    import type { ProVideo } from '../api/pros'
    import type { FrameAnnotation } from '../api/frames'

    const props = defineProps<{
      userVideoUrl: string
      userAnnotations?: FrameAnnotation[]
      proVideo: ProVideo
      timeOffset?: number  // 时间偏移（秒）
    }>()

    const userVideoRef = ref<HTMLVideoElement>()
    const proVideoRef = ref<HTMLVideoElement>()
    const isPlaying = ref(false)
    const playbackRate = ref(1)
    const currentTime = ref(0)
    const showAnnotations = ref(true)

    // 同步播放控制
    function togglePlay() {
      if (isPlaying.value) {
        userVideoRef.value?.pause()
        proVideoRef.value?.pause()
      } else {
        userVideoRef.value?.play()
        proVideoRef.value?.play()
      }
      isPlaying.value = !isPlaying.value
    }

    // 同步速度调整
    watch(playbackRate, (rate) => {
      if (userVideoRef.value) userVideoRef.value.playbackRate = rate
      if (proVideoRef.value) proVideoRef.value.playbackRate = rate
    })

    // 进度条拖动同步
    function seek(time: number) {
      if (userVideoRef.value) userVideoRef.value.currentTime = time
      if (proVideoRef.value) proVideoRef.value.currentTime = time + (props.timeOffset || 0)
    }

    // 监听用户视频时间更新
    function onUserTimeUpdate() {
      if (userVideoRef.value) {
        currentTime.value = userVideoRef.value.currentTime
      }
    }
    </script>

    <template>
      <div class="comparison-player">
        <div class="split-screen flex gap-4">
          <!-- 用户视频 -->
          <div class="user-video flex-1">
            <div class="relative">
              <video
                ref="userVideoRef"
                :src="userVideoUrl"
                @timeupdate="onUserTimeUpdate"
                class="w-full rounded-lg"
              />
              <!-- Canvas 标注层（可选） -->
            </div>
            <div class="text-center mt-2 text-sm text-gray-500">您的视频</div>
          </div>

          <!-- 职业选手视频 -->
          <div class="pro-video flex-1">
            <video
              ref="proVideoRef"
              :src="getProVideoFileUrl(props.proVideo.id)"
              class="w-full rounded-lg"
            />
            <div class="text-center mt-2 text-sm text-gray-500">{{ props.proVideo.name }}</div>
          </div>
        </div>

        <!-- 共享控制栏 -->
        <div class="controls mt-4 flex items-center gap-4">
          <button @click="togglePlay" class="play-btn px-4 py-2 rounded">
            {{ isPlaying ? '暂停' : '播放' }}
          </button>
          <input
            type="range"
            v-model="playbackRate"
            min="0.25"
            max="2"
            step="0.25"
            class="speed-slider"
          />
          <span class="text-sm">{{ playbackRate }}x</span>
          <button @click="showAnnotations = !showAnnotations" class="annotation-btn">
            {{ showAnnotations ? '隐藏标注' : '显示标注' }}
          </button>
        </div>
      </div>
    </template>
    ```

  <name>Task 4: Create Keyframe Marker Component</name>
  <files>web/src/components/KeyframeMarker.vue</files>
  <action>
    创建 `web/src/components/KeyframeMarker.vue`:

    功能: 让用户标记两个视频的关键动作帧（如击球瞬间），自动计算时间偏移。

    实现:
    - Props: userVideoRef, proVideoRef
    - 显示两个视频当前帧的时间
    - 用户可以点击"标记此帧"按钮分别标记两个视频的关键帧
    - 自动计算时间偏移: offset = proFrameTime - userFrameTime
    - emit offset(offsetValue) 供父组件使用
    - 显示对齐状态: "已对齐 ✓" 或 "未对齐"

    ```vue
    <script setup lang="ts">
    import { ref, computed } from 'vue'

    const props = defineProps<{
      userCurrentTime: number
      proCurrentTime: number
    }>()

    const emit = defineEmits<{
      markUser: [time: number]
      markPro: [time: number]
      applyOffset: [offset: number]
    }>()

    const userKeyframe = ref<number | null>(null)
    const proKeyframe = ref<number | null>(null)

    const calculatedOffset = computed(() => {
      if (userKeyframe.value !== null && proKeyframe.value !== null) {
        return proKeyframe.value - userKeyframe.value
      }
      return null
    })

    function markUserFrame() {
      userKeyframe.value = props.userCurrentTime
    }

    function markProFrame() {
      proKeyframe.value = props.proCurrentTime
    }

    function applyOffset() {
      if (calculatedOffset.value !== null) {
        emit('applyOffset', calculatedOffset.value)
      }
    }
    </script>

    <template>
      <div class="keyframe-marker p-4 bg-gray-50 rounded-lg">
        <h4 class="font-semibold mb-3">关键帧对齐</h4>
        <div class="flex gap-4">
          <div class="user-marker">
            <span class="text-sm">用户视频: {{ userCurrentTime.toFixed(2) }}s</span>
            <button @click="markUserFrame" class="ml-2 px-2 py-1 rounded text-sm">
              {{ userKeyframe ? '已标记' : '标记' }}
            </button>
          </div>
          <div class="pro-marker">
            <span class="text-sm">职业视频: {{ proCurrentTime.toFixed(2) }}s</span>
            <button @click="markProFrame" class="ml-2 px-2 py-1 rounded text-sm">
              {{ proKeyframe ? '已标记' : '标记' }}
            </button>
          </div>
        </div>
        <div v-if="calculatedOffset !== null" class="mt-3">
          <span class="text-sm">时间偏移: {{ calculatedOffset.toFixed(2) }}s</span>
          <button @click="applyOffset" class="ml-2 px-3 py-1 bg-blue-500 text-white rounded text-sm">
            应用偏移
          </button>
        </div>
        <div v-else class="mt-3 text-sm text-gray-500">
          请分别标记两个视频的关键动作帧
        </div>
      </div>
    </template>
    ```
